functions {
    print: print a value in the console {
        syntax {
            print(args...)
        }

        example {
            print("hello, world"),

            print("my name is ", "paul")

            print($*varname)
        }
    }

    clear: clear the console {
        syntax {
            clear()
        }
    }
}

keywords {
    var: create a variable {
        syntax {
            var varName = varValue
        }

        example {
            var num = 10

            print($num)
        }

        console {
            10
        }

        we can also reassign a value {
            $num = 5

            print($num)
        }

        console {
            5
        }
    }

    if: conditional structure {
        syntax {
            if (condition == true) {
                code
            }
        }

        example {
            if (2 == 2) {
                print("equals 2")
            }
        }

        operators {
            == 
            !=
            >
            <
            >=
            <=
        }

        console {
            equals 2
        }
    }
}
