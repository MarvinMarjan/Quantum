functions {
    print: print a value in the console {
        syntax {
            print(args...)
        }

        example {
            print("hello, world"),

            print("my name is ", "paul")

            print($*varname)
        }
    }

    clear: clear the console {
        syntax {
            clear()
        }
    }

    ip: ip handling {
        syntax {
            ip(ip_type)
        }

        exmaple {
            var ip = ip("ipv4")

            print($ip)
        }

        console {
            ip*
        }
    }

    file: file manipulation {
        syntax {
            file(mode, path, data?)
        }

        write {
            assigning {
                var fl = file("write", path, initial_data) // returns true if no errors occurred

                print($fl) // true or false
            }

            normal {
                file("write", path, initial_data)
            }
        }

        read {
            assigning {
                var fl = file("read", path) // return the file content

                print($fl)
            }
        }
    }
}

keywords {
    var: create a variable {
        syntax {
            var varName = varValue
        }

        example {
            var num = 10

            print($num)
        }

        console {
            10
        }

        we can also reassign a value {
            $num = 5

            print($num)
        }

        print var type {
            var str = "hello, world"
            var num = 19

            print($Num.type, " ", $str.type)

            console {
                number string
            }
        }

        print var name {
            var num = 10

            print($num.name)

            console {
                num
            }
        }

        conditions {
            var cond = (3 == 3)

            print($cond) // true
        }

        numeric expressions {
            var num = <3 * 2 + (2 + 2)>

            print($num) // 10
        }

        console {
            5
        }
    }

    if: conditional structure {
        syntax {
            if (condition == true) {
                code
            }
        }

        example {
            if (2 == 2) {
                print("equals 2")
            }
        }

        operators {
            == 
            !=
            >
            <
            >=
            <=
        }

        console {
            equals 2
        }
    }

    else: executes a code block if the previous condition is false {
        syntax {
            if (condition) {
                block
            }

            else {
                block
            }
        }

        example {
            if (3 != 3) {
                print("not equals")
            }

            else {
                print("false")
            }
        }

        console {
            false
        }
    }
}
